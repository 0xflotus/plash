usage: plash-docker 
Run the docker client inside alpine linux.



#!/usr/bin/env plash-exec
# plash-exec: exec=/entrypoint root=true
# usage: plash-docker 
# Run the docker client inside alpine linux.

--namespace
docker

--image
alpine

--apk
docker

--layer

--write-script
/entrypoint
#!/bin/sh
export DOCKER_HOST=unix://${PLASH_DOCKER_SOCKET:-/tmp/docker.sock}
exec docker $@
