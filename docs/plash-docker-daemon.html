usage: plash-docker-daemon 
Run the docker daemon inside alpine linux.



#!/usr/bin/env plash-exec
# plash-exec: exec=/entrypoint root=true mount=default,cgroup
# usage: plash-docker-daemon 
# Run the docker daemon inside alpine linux.

--image
alpine

--namespace
docker

--apk
docker

--layer

--write-script
/entrypoint
#!/bin/sh
export DOCKER_RAMDISK=1
exec dockerd -H unix://${PLASH_DOCKER_SOCKET:-/tmp/docker.sock}
